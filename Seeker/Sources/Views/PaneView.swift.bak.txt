import SwiftUI

struct PaneView: View {
    @ObservedObject var pane: PaneState
    @EnvironmentObject var appState: AppState
    let side: AppState.PaneSide

    var isActive: Bool {
        appState.activePane == side
    }

    private var viewModeBinding: Binding<FileExplorerViewModel.ViewMode> {
        Binding(
            get: { pane.activeTab.viewMode },
            set: { pane.activeTab.viewMode = $0 }
        )
    }

    private var searchTextBinding: Binding<String> {
        Binding(
            get: { pane.activeTab.searchText },
            set: { pane.activeTab.searchText = $0 }
        )
    }

    var body: some View {
        VStack(spacing: 0) {
            // Tab bar
            tabBar
            Divider()

            // Path bar + navigation
            paneToolbar
            Divider()

            // File content area
            FileContentView(viewModel: pane.activeTab, side: side)

            Divider()

            // Status bar
            paneStatusBar
        }
        .background(isActive ? Color.clear : Color(nsColor: .windowBackgroundColor).opacity(0.5))
        .overlay(
            Rectangle()
                .stroke(isActive ? Color.accentColor.opacity(0.6) : Color.clear, lineWidth: 2)
        )
        .contentShape(Rectangle())
        .onTapGesture {
            appState.activePane = side
        }
    }

    // MARK: - Tab Bar

    private var tabBar: some View {
        HStack(spacing: 0) {
            ScrollView(.horizontal, showsIndicators: false) {
                HStack(spacing: 0) {
                    ForEach(Array(pane.tabs.enumerated()), id: \.element.id) { index, tab in
                        TabItemView(
                            title: tab.tabTitle,
                            isSelected: index == pane.activeTabIndex,
                            onSelect: {
                                pane.selectTab(index)
                                appState.activePane = side
                            },
                            onClose: pane.tabs.count > 1 ? {
                                pane.closeTab(at: index)
                            } : nil
                        )
                    }
                }
            }

            Spacer()

            // New tab button
            Button {
                pane.addTab()
                appState.activePane = side
            } label: {
                Image(systemName: "plus")
                    .font(.caption)
                    .foregroundColor(.secondary)
            }
            .buttonStyle(.borderless)
            .padding(.horizontal, 8)
        }
        .frame(height: 28)
        .background(Color(nsColor: .controlBackgroundColor))
    }

    // MARK: - Pane Toolbar (nav + path + view mode)

    private var paneToolbar: some View {
        HStack(spacing: 6) {
            // Navigation
            Button(action: pane.activeTab.goBack) {
                Image(systemName: "chevron.left")
            }
            .disabled(!pane.activeTab.canGoBack)
            .buttonStyle(.borderless)

            Button(action: pane.activeTab.goForward) {
                Image(systemName: "chevron.right")
            }
            .disabled(!pane.activeTab.canGoForward)
            .buttonStyle(.borderless)

            Button(action: pane.activeTab.goUp) {
                Image(systemName: "chevron.up")
            }
            .disabled(!pane.activeTab.canGoUp)
            .buttonStyle(.borderless)

            // Path breadcrumb
            ScrollView(.horizontal, showsIndicators: false) {
                HStack(spacing: 2) {
                    let components = pane.activeTab.pathComponents
                    ForEach(0..<components.count, id: \.self) { index in
                        if index > 0 {
                            Image(systemName: "chevron.right")
                                .font(.system(size: 8))
                                .foregroundColor(.secondary.opacity(0.5))
                        }
                        Button(components[index].0) {
                            pane.activeTab.navigateTo(components[index].1)
                        }
                        .buttonStyle(.borderless)
                        .font(.system(size: 11))
                        .foregroundColor(index == components.count - 1 ? .primary : .secondary)
                    }
                }
            }

            // View mode picker
            Picker("", selection: viewModeBinding) {
                Image(systemName: "list.bullet").tag(FileExplorerViewModel.ViewMode.list)
                Image(systemName: "square.grid.2x2").tag(FileExplorerViewModel.ViewMode.icons)
                Image(systemName: "rectangle.split.3x1").tag(FileExplorerViewModel.ViewMode.columns)
            }
            .pickerStyle(.segmented)
            .frame(width: 100)

            // Filter
            HStack(spacing: 3) {
                Image(systemName: "magnifyingglass")
                    .font(.system(size: 10))
                    .foregroundColor(.secondary)
                TextField("Filter", text: searchTextBinding)
                    .textFieldStyle(.plain)
                    .font(.system(size: 11))
                    .onChange(of: pane.activeTab.searchText) { _ in
                        pane.activeTab.loadFiles()
                    }
                if !pane.activeTab.searchText.isEmpty {
                    Button {
                        pane.activeTab.searchText = ""
                        pane.activeTab.loadFiles()
                    } label: {
                        Image(systemName: "xmark.circle.fill")
                            .font(.system(size: 10))
                            .foregroundColor(.secondary)
                    }
                    .buttonStyle(.borderless)
                }
            }
            .padding(.horizontal, 5)
            .padding(.vertical, 2)
            .background(Color(nsColor: .controlBackgroundColor))
            .cornerRadius(5)
            .frame(width: 140)
        }
        .padding(.horizontal, 8)
        .padding(.vertical, 4)
    }

    // MARK: - Status Bar

    private var paneStatusBar: some View {
        HStack(spacing: 8) {
            Text("\(pane.activeTab.files.count) items")
                .font(.system(size: 10))
                .foregroundColor(.secondary)

            if let sel = pane.activeTab.selectedFile {
                Text("·")
                    .foregroundColor(.secondary.opacity(0.5))
                Text(sel.name)
                    .font(.system(size: 10))
                    .foregroundColor(.secondary)
                    .lineLimit(1)
                if !sel.isDirectory {
                    Text("(\(sel.formattedSize))")
                        .font(.system(size: 10))
                        .foregroundColor(.secondary.opacity(0.7))
                }
            }

            Spacer()

            // Free space
            if let space = freeSpace() {
                Text(space)
                    .font(.system(size: 10))
                    .foregroundColor(.secondary.opacity(0.7))
            }
        }
        .padding(.horizontal, 8)
        .padding(.vertical, 3)
        .background(Color(nsColor: .controlBackgroundColor))
    }

    private func freeSpace() -> String? {
        guard let values = try? pane.activeTab.currentURL.resourceValues(forKeys: [.volumeAvailableCapacityForImportantUsageKey]),
              let bytes = values.volumeAvailableCapacityForImportantUsage else { return nil }
        return ByteCountFormatter.string(fromByteCount: bytes, countStyle: .file) + " free"
    }
}

// MARK: - Tab Item

struct TabItemView: View {
    let title: String
    let isSelected: Bool
    let onSelect: () -> Void
    let onClose: (() -> Void)?

    @State private var isHovering = false

    var body: some View {
        HStack(spacing: 4) {
            Text(title)
                .font(.system(size: 11))
                .lineLimit(1)
                .foregroundColor(isSelected ? .primary : .secondary)

            if let onClose = onClose, (isHovering || isSelected) {
                Button(action: onClose) {
                    Image(systemName: "xmark")
                        .font(.system(size: 8, weight: .bold))
                        .foregroundColor(.secondary)
                }
                .buttonStyle(.borderless)
            }
        }
        .padding(.horizontal, 12)
        .padding(.vertical, 4)
        .background(isSelected ? Color(nsColor: .controlBackgroundColor) : Color.clear)
        .overlay(
            VStack {
                if isSelected {
                    Rectangle()
                        .fill(Color.accentColor)
                        .frame(height: 2)
                }
                Spacer()
            }
        )
        .contentShape(Rectangle())
        .onTapGesture(perform: onSelect)
        .onHover { isHovering = $0 }
    }
}
